STL容器基本概念

什么是容器

容器是用来存储和管理数据的对象，STL提供了多种不同特性的容器以满足不同的需求。

容器的分类

  • 顺序容器：元素按线性顺序排列
    • vector：动态数组
    • list：双向链表
    • deque：双端队列
  • 关联容器：通过键值对存储数据
    • set：集合
    • map：映射
    • multiset：多重集合
    • multimap：多重映射
  • 容器适配器：基于其他容器实现特定接口
    • stack：栈
    • queue：队列
    • priority_queue：优先队列

常用容器对比

不同容器有各自的特点和适用场景：

容器 底层实现 插入/删除位置 访问方式 特点
vector 动态数组 尾部高效 随机访问 内存连续，支持下标访问
stack 默认deque 顶部 LIFO 后进先出
queue 默认deque 尾部插入，首部删除 FIFO 先进先出
priority_queue 默认vector 顶部 优先级最高元素 自动排序

vector（动态数组）

vector的特点

vector是STL中最常用的容器之一，它是一个能够动态增长的数组：

  • 内存连续存储
  • 支持随机访问（通过下标）
  • 在尾部插入和删除元素效率高
  • 在中间插入或删除元素效率较低

vector的基本操作

创建和初始化vector

#include <iostream>
#include <vector>
using namespace std;

int main() {
    // 创建空的vector
    vector<int> v1;

    // 创建指定大小的vector，默认值为0
    vector<int> v2(5);

    // 创建指定大小和初始值的vector
    vector<int> v3(5, 100);

    // 通过初始化列表创建vector
    vector<int> v4 = {1, 2, 3, 4, 5};

    // 通过另一个vector创建
    vector<int> v5(v4);

    return 0;
}

vector的常用操作

#include <iostream>
#include <vector>
using namespace std;

int main() {
    vector<int> vec = {1, 2, 3, 4, 5};

    // 添加元素
    vec.push_back(6);  // 在尾部添加元素

    // 访问元素
    cout << "第一个元素: " << vec[0] << endl;
    cout << "最后一个元素: " << vec.back() << endl;

    // 获取容器信息
    cout << "元素个数: " << vec.size() << endl;
    cout << "是否为空: " << (vec.empty() ? "是" : "否") << endl;

    // 修改元素
    vec[0] = 10;

    // 删除元素
    vec.pop_back();  // 删除最后一个元素

    // 遍历vector
    cout << "所有元素: ";
    for (int i = 0; i < vec.size(); i++) {
        cout << vec[i] << " ";
    }
    cout << endl;

    // 使用迭代器遍历
    cout << "使用迭代器: ";
    for (vector<int>::iterator it = vec.begin(); it != vec.end(); ++it) {
        cout << *it << " ";
    }
    cout << endl;

    return 0;
}

stack（栈）

stack的特点

stack是一种容器适配器，它实现了栈的数据结构：

  • 后进先出（LIFO）的数据结构
  • 只能访问栈顶元素
  • 只能在栈顶插入或删除元素
  • 默认基于deque实现

stack的基本操作

stack的基本使用

#include <iostream>
#include <stack>
using namespace std;

int main() {
    // 创建stack
    stack<int> stk;

    // 判断是否为空
    if (stk.empty()) {
        cout << "栈为空" << endl;
    }

    // 入栈操作
    stk.push(10);
    stk.push(20);
    stk.push(30);

    // 获取栈的大小
    cout << "栈的大小: " << stk.size() << endl;

    // 访问栈顶元素
    cout << "栈顶元素: " << stk.top() << endl;

    // 出栈操作
    stk.pop();
    cout << "出栈后栈顶元素: " << stk.top() << endl;

    // 清空栈
    while (!stk.empty()) {
        cout << "弹出: " << stk.top() << endl;
        stk.pop();
    }

    return 0;
}

stack应用示例：括号匹配检查

#include <iostream>
#include <stack>
#include <string>
using namespace std;

bool isBalanced(const string& expr) {
    stack<char> s;

    for (char c : expr) {
        if (c == '(' || c == '[' || c == '{') {
            // 遇到开括号，入栈
            s.push(c);
        } else if (c == ')' || c == ']' || c == '}') {
            // 遇到闭括号
            if (s.empty()) {
                // 栈为空，不匹配
                return false;
            }

            char top = s.top();
            s.pop();

            // 检查括号是否匹配
            if ((c == ')' && top != '(') ||
                (c == ']' && top != '[') ||
                (c == '}' && top != '{')) {
                return false;
            }
        }
    }

    // 检查栈是否为空
    return s.empty();
}

int main() {
    string expr1 = "{[()()]}";
    string expr2 = "{[()]}";
    string expr3 = "{[()])";

    cout << expr1 << " 是否匹配: " << (isBalanced(expr1) ? "是" : "否") << endl;
    cout << expr2 << " 是否匹配: " << (isBalanced(expr2) ? "是" : "否") << endl;
    cout << expr3 << " 是否匹配: " << (isBalanced(expr3) ? "是" : "否") << endl;

    return 0;
}

queue（队列）

queue的特点

queue是一种容器适配器，它实现了队列的数据结构：

  • 先进先出（FIFO）的数据结构
  • 只能访问队首和队尾元素
  • 在队尾插入元素，在队首删除元素
  • 默认基于deque实现

queue的基本操作

queue的基本使用

#include <iostream>
#include <queue>
using namespace std;

int main() {
    // 创建queue
    queue<int> q;

    // 判断是否为空
    if (q.empty()) {
        cout << "队列为空" << endl;
    }

    // 入队操作
    q.push(10);
    q.push(20);
    q.push(30);

    // 获取队列的大小
    cout << "队列的大小: " << q.size() << endl;

    // 访问队首和队尾元素
    cout << "队首元素: " << q.front() << endl;
    cout << "队尾元素: " << q.back() << endl;

    // 出队操作
    q.pop();
    cout << "出队后队首元素: " << q.front() << endl;

    // 清空队列
    while (!q.empty()) {
        cout << "出队: " << q.front() << endl;
        q.pop();
    }

    return 0;
}

priority_queue（优先队列）

priority_queue的特点

priority_queue是一种容器适配器，它实现了优先队列的数据结构：

  • 元素按优先级排序
  • 默认情况下，最大的元素优先级最高
  • 只能访问优先级最高的元素
  • 默认基于vector实现，使用堆算法维护顺序

priority_queue的基本操作

priority_queue的基本使用

#include <iostream>
#include <queue>
using namespace std;

int main() {
    // 创建priority_queue（默认大顶堆）
    priority_queue<int> pq;

    // 判断是否为空
    if (pq.empty()) {
        cout << "优先队列为空" << endl;
    }

    // 插入元素
    pq.push(30);
    pq.push(10);
    pq.push(50);
    pq.push(20);
    pq.push(40);

    // 获取优先队列的大小
    cout << "优先队列的大小: " << pq.size() << endl;

    // 访问优先级最高的元素
    cout << "优先级最高的元素: " << pq.top() << endl;

    // 删除优先级最高的元素
    pq.pop();
    cout << "删除后优先级最高的元素: " << pq.top() << endl;

    // 输出所有元素（按优先级顺序）
    cout << "按优先级输出: ";
    while (!pq.empty()) {
        cout << pq.top() << " ";
        pq.pop();
    }
    cout << endl;

    return 0;
}

priority_queue应用示例：任务调度（按优先级）

#include <iostream>
#include <queue>
#include <string>
using namespace std;

// 定义任务结构
struct Task {
    int priority;
    string name;

    // 重载小于运算符，实现按优先级排序
    bool operator<(const Task& other) const {
        return priority < other.priority;  // 优先级高的先执行
    }
};

int main() {
    // 创建任务优先队列
    priority_queue<Task> task_queue;

    // 添加任务
    task_queue.push({1, "普通任务1"});
    task_queue.push({3, "紧急任务1"});
    task_queue.push({2, "重要任务1"});
    task_queue.push({5, "紧急任务2"});
    task_queue.push({1, "普通任务2"});

    // 执行任务
    cout << "按优先级执行任务:" << endl;
    while (!task_queue.empty()) {
        Task current = task_queue.top();
        cout << "执行 [" << current.name << "] 优先级: " << current.priority << endl;
        task_queue.pop();
    }

    return 0;
}

练习1：使用vector管理学生成绩

编写程序使用vector存储和管理学生成绩：

#include <iostream>
#include <vector>
using namespace std;

int main() {
    // 创建存储学生成绩的vector
    vector<int> scores;

    // 输入成绩
    int n, score;
    cout << "请输入学生人数: ";
    cin >> n;

    cout << "请输入学生成绩: ";
    for (int i = 0; i < n; i++) {
        cin >> score;
        scores.push_back(score);
    }

    // 输出所有成绩
    cout << "学生成绩: ";
    for (int i = 0; i < scores.size(); i++) {
        cout << scores[i] << " ";
    }
    cout << endl;

    // 计算平均分
    int sum = 0;
    for (int i = 0; i < scores.size(); i++) {
        sum += scores[i];
    }
    double average = (double)sum / scores.size();
    cout << "平均分: " << average << endl;

    // 查找最高分和最低分
    int max_score = scores[0], min_score = scores[0];
    for (int i = 1; i < scores.size(); i++) {
        if (scores[i] > max_score) max_score = scores[i];
        if (scores[i] < min_score) min_score = scores[i];
    }
    cout << "最高分: " << max_score << endl;
    cout << "最低分: " << min_score << endl;

    return 0;
}

练习2：使用stack实现表达式求值

使用两个stack分别存储操作数和运算符，实现简单表达式求值：

#include <iostream>
#include <stack>
#include <string>
using namespace std;

int main() {
    // 简化版本，仅处理个位数的加减乘除运算
    string expression = "3+5*2";
    stack<int> numbers;
    stack<char> operators;

    // 解析表达式
    for (int i = 0; i < expression.length(); i++) {
        char c = expression[i];

        if (isdigit(c)) {
            // 数字直接入栈
            numbers.push(c - '0');
        } else {
            // 运算符处理
            if (!operators.empty() &&
                (c == '+' || c == '-') &&
                (operators.top() == '*' || operators.top() == '/')) {
                // 如果当前运算符优先级低于栈顶运算符，先计算栈顶运算
                char op = operators.top();
                operators.pop();

                int b = numbers.top();
                numbers.pop();
                int a = numbers.top();
                numbers.pop();

                if (op == '*') numbers.push(a * b);
                if (op == '/') numbers.push(a / b);
            }
            operators.push(c);
        }
    }

    // 处理剩余运算
    while (!operators.empty()) {
        char op = operators.top();
        operators.pop();

        int b = numbers.top();
        numbers.pop();
        int a = numbers.top();
        numbers.pop();

        if (op == '+') numbers.push(a + b);
        if (op == '-') numbers.push(a - b);
        if (op == '*') numbers.push(a * b);
        if (op == '/') numbers.push(a / b);
    }

    // 输出结果
    cout << expression << " = " << numbers.top() << endl;

    return 0;
}

练习4：使用priority_queue实现成绩排名

使用priority_queue对学生按成绩进行排名：

#include <iostream>
#include <queue>
#include <string>
using namespace std;

// 定义学生结构
struct Student {
    string name;
    int score;

    // 重载小于运算符，实现按成绩排序
    bool operator<(const Student& other) const {
        return score < other.score;  // 分数高的优先级高
    }
};

int main() {
    // 创建学生优先队列
    priority_queue<Student> students;

    // 添加学生信息
    students.push({"张三", 85});
    students.push({"李四", 92});
    students.push({"王五", 78});
    students.push({"赵六", 96});
    students.push({"钱七", 88});

    // 按排名输出
    cout << "学生成绩排名:" << endl;
    int rank = 1;
    while (!students.empty()) {
        Student s = students.top();
        cout << "第" << rank << "名: " << s.name << " 分数: " << s.score << endl;
        students.pop();
        rank++;
    }

    return 0;
}
