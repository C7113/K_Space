Python二级课程第五课

字典与复杂数据结构

课程目标

掌握Python字典的基本概念、特点和常用操作，理解复杂数据结构的构建和应用，能够熟练使用字典进行数据存储和处理。

一、字典基础

1. 字典的概念与特点

字典是Python中一种可变容器模型，可存储任意类型对象，以键值对的形式存储数据。

  • 键值对结构：每个元素由键(key)和值(value)组成
  • 无序集合：元素没有固定顺序（Python 3.7+有序）
  • 可变长度：可以动态添加、删除键值对
  • 键必须唯一：同一个字典中键不能重复
  • 键必须不可变：键可以是字符串、数字、元组等不可变类型
# 字典的特点示例
# 字典定义
student = {"name": "张三", "age": 18, "score": 90}
print(student) # 输出：{'name': '张三', 'age': 18, 'score': 90}

# 键必须是不可变类型
valid_keys = {
    "name": "字符串键",
    123: "整数键",
    (1, 2): "元组键"
}
print(valid_keys) # 输出：{'name': '字符串键', 123: '整数键', (1, 2): '元组键'}

# 键不能重复，后面的会覆盖前面的
duplicate_keys = {"a": 1, "a": 2}
print(duplicate_keys) # 输出：{'a': 2}

# 错误示例：列表不能作为键
# invalid_dict = {[1,2]: "value"} # 这行会报错：TypeError

2. 字典的声明方法

声明方法 示例 说明
直接声明 d = {"key": "value"} 使用花括号和冒号
dict()函数 d = dict(key="value") 关键字参数形式
dict()函数 d = dict([("key", "value")]) 键值对元组列表
fromkeys() d = dict.fromkeys(["k1","k2"]) 创建具有相同值的字典
# 字典声明示例
# 方法1：直接声明
dict1 = {"name": "Alice", "age": 20}
print("直接声明:", dict1) # 输出：直接声明: {'name': 'Alice', 'age': 20}

# 方法2：dict()函数（关键字参数）
dict2 = dict(name="Bob", age=25)
print("dict()关键字:", dict2) # 输出：dict()关键字: {'name': 'Bob', 'age': 25}

# 方法3：dict()函数（元组列表）
dict3 = dict([("name", "Charlie"), ("age", 30)])
print("dict()元组:", dict3) # 输出：dict()元组: {'name': 'Charlie', 'age': 30}

# 方法4：fromkeys()方法
dict4 = dict.fromkeys(["name", "age", "score"])
print("fromkeys()默认:", dict4) # 输出：fromkeys()默认: {'name': None, 'age': None, 'score': None}

dict5 = dict.fromkeys(["name", "age", "score"], "unknown")
print("fromkeys()指定值:", dict5) # 输出：fromkeys()指定值: {'name': 'unknown', 'age': 'unknown', 'score': 'unknown'}

# 空字典
empty_dict1 = {}
empty_dict2 = dict()
print("空字典1:", empty_dict1) # 输出：空字典1: {}
print("空字典2:", empty_dict2) # 输出：空字典2: {}

二、字典常用操作

1. 增加和修改元素

方法 说明 示例
字典[键] = 值 添加或修改键值对 d["key"] = "value"
setdefault(键, 值) 键不存在时添加，存在时不修改 d.setdefault("k", "v")
update(字典2) 用字典2更新当前字典 d1.update(d2)
# 增加和修改元素示例
student = {"name": "张三", "age": 18}
print("原字典:", student)

# 添加新键值对
student["score"] = 90
print("添加score:", student) # 输出：添加score: {'name': '张三', 'age': 18, 'score': 90}

# 修改已有键的值
student["age"] = 19
print("修改age:", student) # 输出：修改age: {'name': '张三', 'age': 19, 'score': 90}

# 使用setdefault添加（键不存在时）
student.setdefault("gender", "男")
print("setdefault添加:", student) # 输出：setdefault添加: {'name': '张三', 'age': 19, 'score': 90, 'gender': '男'}

# 使用setdefault（键存在时不修改）
student.setdefault("name", "李四")
print("setdefault存在:", student) # 输出：setdefault存在: {'name': '张三', 'age': 19, 'score': 90, 'gender': '男'}

# 使用update更新字典
update_info = {"age": 20, "class": "一班"}
student.update(update_info)
print("update更新:", student) # 输出：update更新: {'name': '张三', 'age': 20, 'score': 90, 'gender': '男', 'class': '一班'}

2. 删除元素

方法 说明 示例
del 字典[键] 删除指定键值对 del d["key"]
pop(键) 删除并返回指定键的值 d.pop("key")
popitem() 删除并返回最后一个键值对 d.popitem()
clear() 清空字典 d.clear()
# 删除元素示例
student = {"name": "张三", "age": 18, "score": 90, "class": "一班"}
print("原字典:", student)

# 使用del删除
del student["class"]
print("del删除class:", student) # 输出：del删除class: {'name': '张三', 'age': 18, 'score': 90}

# 使用pop删除并返回值
age = student.pop("age")
print("pop删除age, 返回值:", age) # 输出：pop删除age, 返回值: 18
print("pop后字典:", student) # 输出：pop后字典: {'name': '张三', 'score': 90}

# 使用popitem删除最后一个元素
last_item = student.popitem()
print("popitem返回值:", last_item) # 输出：popitem返回值: ('score', 90)
print("popitem后字典:", student) # 输出：popitem后字典: {'name': '张三'}

# 使用clear清空字典
student.clear()
print("clear后字典:", student) # 输出：clear后字典: {}

3. 查询元素

方法 说明 示例
字典[键] 获取指定键的值，键不存在会报错 d["key"]
get(键) 获取指定键的值，键不存在返回None d.get("key")
get(键, 默认值) 获取指定键的值，键不存在返回默认值 d.get("key", "default")
keys() 返回所有键的视图 d.keys()
values() 返回所有值的视图 d.values()
items() 返回所有键值对的视图 d.items()
# 查询元素示例
student = {"name": "张三", "age": 18, "score": 90}
print("原字典:", student)

# 使用键直接访问
name = student["name"]
print("直接访问name:", name) # 输出：直接访问name: 张三

# 使用get方法访问（安全）
age = student.get("age")
print("get访问age:", age) # 输出：get访问age: 18

# 访问不存在的键
# gender = student["gender"] # 这行会报错：KeyError
gender = student.get("gender")
print("get访问不存在的键:", gender) # 输出：get访问不存在的键: None

gender = student.get("gender", "未知")
print("get访问不存在的键(带默认值):", gender) # 输出：get访问不存在的键(带默认值): 未知

# 获取所有键
keys = student.keys()
print("所有键:", keys) # 输出：所有键: dict_keys(['name', 'age', 'score'])

# 获取所有值
values = student.values()
print("所有值:", values) # 输出：所有值: dict_values(['张三', 18, 90])

# 获取所有键值对
items = student.items()
print("所有键值对:", items) # 输出：所有键值对: dict_items([('name', '张三'), ('age', 18), ('score', 90)])

4. 字典遍历

# 字典遍历示例
student = {"name": "张三", "age": 18, "score": 90}

# 遍历所有键
print("遍历所有键:")
for key in student.keys():
    print(f"键: {key}")

# 遍历所有值
print("\n遍历所有值:")
for value in student.values():
    print(f"值: {value}")

# 遍历所有键值对
print("\n遍历所有键值对:")
for key, value in student.items():
    print(f"{key}: {value}")

# 直接遍历字典（默认遍历键）
print("\n直接遍历字典(键):")
for key in student:
    print(f"键: {key}, 值: {student[key]}")

# 输出结果：
# 遍历所有键:
# 键: name
# 键: age
# 键: score
#
# 遍历所有值:
# 值: 张三
# 值: 18
# 值: 90
#
# 遍历所有键值对:
# name: 张三
# age: 18
# score: 90
#
# 直接遍历字典(键):
# 键: name, 值: 张三
# 键: age, 值: 18
# 键: score, 值: 90

5. 字典复制

# 字典复制示例
original = {"name": "张三", "age": 18, "scores": [90, 85, 95]}

# 直接赋值（引用同一个对象）
reference = original
reference["name"] = "李四"
print("原字典:", original) # 输出：原字典: {'name': '李四', 'age': 18, 'scores': [90, 85, 95]}

# 重置原字典
original = {"name": "张三", "age": 18, "scores": [90, 85, 95]}

# 浅拷贝
shallow_copy = original.copy()
shallow_copy["name"] = "王五"
shallow_copy["scores"].append(100) # 修改嵌套对象
print("原字典:", original) # 输出：原字典: {'name': '张三', 'age': 18, 'scores': [90, 85, 95, 100]}
print("浅拷贝:", shallow_copy) # 输出：浅拷贝: {'name': '王五', 'age': 18, 'scores': [90, 85, 95, 100]}

# 深拷贝
import copy
original = {"name": "张三", "age": 18, "scores": [90, 85, 95]}
deep_copy = copy.deepcopy(original)
deep_copy["scores"].append(100)
print("原字典:", original) # 输出：原字典: {'name': '张三', 'age': 18, 'scores': [90, 85, 95]}
print("深拷贝:", deep_copy) # 输出：深拷贝: {'name': '张三', 'age': 18, 'scores': [90, 85, 95, 100]}
注意：浅拷贝只复制第一层，嵌套对象仍然是引用。深拷贝会递归复制所有嵌套对象。

三、复杂数据结构

1. 字典的嵌套

字典的值可以是任意类型，包括列表、字典等，这样可以构建复杂的数据结构。

# 字典嵌套示例
# 字典中嵌套列表
student1 = {
    "name": "张三",
    "age": 18,
    "scores": [90, 85, 95]
}

# 字典中嵌套字典
student2 = {
    "name": "李四",
    "age": 19,
    "contact": {
        "phone": "123456789",
        "email": "lisi@example.com"
    }
}

# 列表中嵌套字典
students = [
    {"name": "张三", "age": 18},
    {"name": "李四", "age": 19},
    {"name": "王五", "age": 20}
]

print("学生1:", student1)
print("学生2:", student2)
print("学生列表:", students)

# 访问嵌套数据
print("张三的第一个成绩:", student1["scores"][0]) # 输出：张三的第一个成绩: 90
print("李四的电话:", student2["contact"]["phone"]) # 输出：李四的电话: 123456789
print("第二个学生的名字:", students[1]["name"]) # 输出：第二个学生的名字: 李四

2. 复杂数据结构的应用

# 复杂数据结构应用示例
# 学校数据结构
school = {
    "name": "第一中学",
    "classes": [
        {
            "class_name": "高一(1)班",
            "teacher": "张老师",
            "students": [
                {"name": "张三", "age": 16, "scores": {"math": 90, "english": 85}},
                {"name": "李四", "age": 17, "scores": {"math": 95, "english": 88}}
            ]
        },
        {
            "class_name": "高一(2)班",
            "teacher": "李老师",
            "students": [
                {"name": "王五", "age": 16, "scores": {"math": 88, "english": 92}}
            ]
        }
    ]
}

# 访问复杂数据
print("学校名称:", school["name"])
print("第一个班级:", school["classes"][0]["class_name"])
print("第一个班级的班主任:", school["classes"][0]["teacher"])
print("第一个班级第一个学生:", school["classes"][0]["students"][0]["name"])
print("张三的数学成绩:", school["classes"][0]["students"][0]["scores"]["math"])

# 遍历所有班级和学生
for class_info in school["classes"]:
    print(f"\n{class_info['class_name']} - {class_info['teacher']}")
    for student in class_info["students"]:
        print(f" - {student['name']}, 年龄: {student['age']}")

四、字典内置函数

常用内置函数

函数 说明 示例
len(dict) 返回字典中键值对的数量 len({"a":1}) → 1
str(dict) 将字典转换为字符串表示 str({"a":1}) → "{'a': 1}"
type(dict) 返回对象的类型 type({}) → <class 'dict'>
# 字典内置函数示例
student = {"name": "张三", "age": 18, "score": 90}

# 获取字典长度
length = len(student)
print("字典长度:", length) # 输出：字典长度: 3

# 转换为字符串
str_dict = str(student)
print("字符串表示:", str_dict) # 输出：字符串表示: {'name': '张三', 'age': 18, 'score': 90}
print("类型:", type(str_dict)) # 输出：类型: <class 'str'>

# 检查类型
print("字典类型:", type(student)) # 输出：字典类型: <class 'dict'>

# 检查键是否存在
has_name = "name" in student
has_gender = "gender" in student
print("是否有name:", has_name) # 输出：是否有name: True
print("是否有gender:", has_gender) # 输出：是否有gender: False

创建一个学生字典：student = {"name": "张三", "age": 18, "major": "计算机"}

完成以下操作：

  1. 添加键值对 "grade": 90

题目2：字典综合应用

创建一个班级字典，包含多个学生的信息：

  1. 创建一个空字典class_info
  2. 添加班级名称"class_name": "Python二级班"
  3. 添加学生列表"students"，包含至少3个学生字典
  4. 计算班级学生的平均年龄
  5. 找出年龄最大的学生
  6. 统计每个专业的学生人数
# 答案
# 1. 创建空字典
class_info = {}

# 2. 添加班级名称
class_info["class_name"] = "Python二级班"

# 3. 添加学生列表
class_info["students"] = [
    {"name": "张三", "age": 18, "major": "计算机"},
    {"name": "李四", "age": 19, "major": "数学"},
    {"name": "王五", "age": 20, "major": "计算机"},
    {"name": "赵六", "age": 18, "major": "物理"}
]

print("班级信息:", class_info)

# 4. 计算平均年龄
total_age = 0
for student in class_info["students"]:
    total_age += student["age"]
average_age = total_age / len(class_info["students"])
print(f"平均年龄: {average_age:.1f}")

# 5. 找出年龄最大的学生
oldest_student = max(class_info["students"], key=lambda x: x["age"])
print(f"年龄最大的学生: {oldest_student['name']}, 年龄: {oldest_student['age']}")

# 6. 统计每个专业的学生人数
major_count = {}
for student in class_info["students"]:
    major = student["major"]
    major_count[major] = major_count.get(major, 0) + 1

print("专业分布:")
for major, count in major_count.items():
    print(f"{major}: {count}人")

题目3：复杂数据结构操作

给定一个嵌套字典，表示学校信息：

school = {
    "name": "第一中学",
    "classes": [
        {
            "name": "高一(1)班",

            "students": [
                {"name": "张三", "scores": {"语文": 85, "数学": 90, "英语": 88}},

        },
        {
            "name": "高一(2)班",
            "teacher": "李老师",
            "students": [
                {"name": "王五", "scores": {"语文": 88, "数学": 92, "英语": 85}}

        }
    ]
}

完成以下操作：

  1. 获取"高一(1)班"班主任的名字
  2. 获取"李四"的数学成绩
  3. 计算"高一(1)班"学生的数学平均分
  4. 为"高一(2)班"添加一个新学生
  5. 找出全校数学成绩最高的学生
# 答案
school = {
    "name": "第一中学",
    "classes": [
        {
            "name": "高一(1)班",
            "teacher": "张老师",
            "students": [
                {"name": "张三", "scores": {"语文": 85, "数学": 90, "英语": 88}},
                {"name": "李四", "scores": {"语文": 92, "数学": 95, "英语": 90}}
            ]
        },
        {
            "name": "高一(2)班",
            "teacher": "李老师",
            "students": [
                {"name": "王五", "scores": {"语文": 88, "数学": 92, "英语": 85}}
            ]
        }
    ]
}

# 1. 获取"高一(1)班"班主任的名字
teacher_name = school["classes"][0]["teacher"]
print("高一(1)班班主任:", teacher_name)

# 2. 获取"李四"的数学成绩
lisi_math = school["classes"][0]["students"][1]["scores"]["数学"]
print("李四的数学成绩:", lisi_math)

# 3. 计算"高一(1)班"学生的数学平均分
class1_students = school["classes"][0]["students"]
math_scores = [student["scores"]["数学"] for student in class1_students]
average_math = sum(math_scores) / len(math_scores)
print("高一(1)班数学平均分:", average_math)

# 4. 为"高一(2)班"添加一个新学生
new_student = {"name": "赵六", "scores": {"语文": 90, "数学": 88, "英语": 92}}
school["classes"][1]["students"].append(new_student)
print("添加新学生后高一(2)班:")
for student in school["classes"][1]["students"]:
    print(f" - {student['name']}")

# 5. 找出全校数学成绩最高的学生
best_math_score = 0
best_math_student = ""

for class_info in school["classes"]:
    for student in class_info["students"]:
        if student["scores"]["数学"] > best_math_score:
            best_math_score = student["scores"]["数学"]
            best_math_student = student["name"]

print(f"数学成绩最高的学生: {best_math_student}, 成绩: {best_math_score}")

课程要点

字典核心概念

  • 键值对结构
  • 无序集合（Python 3.7+有序）
  • 键必须唯一且不可变
  • 值可以是任意类型

重要字典方法

  • 增改：直接赋值、setdefault、update
  • 删除：del、pop、popitem、clear
  • 查询：直接访问、get、keys、values、items
  • 复制：copy、deepcopy

字典遍历

  • 遍历键：for key in dict
  • 遍历值：for value in dict.values()
  • 遍历键值对：for key, value in dict.items()

复杂数据结构

  • 字典嵌套列表
  • 字典嵌套字典
  • 列表中嵌套字典
  • 多层嵌套结构

常见错误

  • 使用可变对象作为键
  • 直接访问不存在的键
  • 混淆浅拷贝和深拷贝
  • 在遍历时修改字典
