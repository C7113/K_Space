一、计算思维与问题解决

1. 计算思维的概念

计算思维是一种运用计算机科学基本概念进行问题求解、系统设计和人类行为理解的思维方式。

  • 分解：将复杂问题分解为更小、更易管理的部分
  • 模式识别：发现问题的规律和模式
  • 抽象：提取问题的核心特征，忽略无关细节
  • 算法设计：设计步骤清晰的解决方案
# 计算思维示例：计算平均分
# 1. 分解：输入成绩 -> 计算总和 -> 计算平均 -> 输出结果
# 2. 模式识别：所有成绩都需要相加，然后除以数量
# 3. 抽象：只关心成绩数值，不关心具体学生
# 4. 算法设计：明确的步骤

scores = [85, 92, 78, 90, 88]
total = 0
count = 0

for score in scores:
    total += score
    count += 1

average = total / count
print(f"平均分: {average:.2f}")

2. 问题解决流程

# 问题解决流程示例：找出列表中的最大值
# 1. 理解问题：需要在列表中找出最大的数字
# 2. 制定计划：遍历列表，比较每个元素
# 3. 执行计划：编写代码实现
# 4. 检查回顾：验证结果是否正确

numbers = [3, 7, 2, 9, 1, 8, 5]

# 初始化最大值
max_value = numbers[0]

# 遍历列表，比较每个元素
for num in numbers:
    if num > max_value:
        max_value = num

print(f"列表中的最大值: {max_value}")

# 检查：使用内置函数验证
print(f"验证: {max(numbers)}")

1. 学生成绩管理系统

# 学生成绩管理系统
def display_menu():
    print("\n=== 学生成绩管理系统 ===")
    print("1. 添加学生成绩")
    print("2. 查看所有成绩")
    print("3. 计算平均分")
    print("4. 查找学生成绩")
    print("5. 删除学生成绩")
    print("6. 退出系统")

def add_student(students):
    name = input("请输入学生姓名: ")
    score = float(input("请输入学生成绩: "))
    students[name] = score
    print(f"成功添加学生 {name}，成绩 {score}")

def view_all_students(students):
    if not students:
        print("暂无学生成绩记录")
        return
    
    print("\n所有学生成绩:")
    for name, score in students.items():
        print(f"{name}: {score}")

def calculate_average(students):
    if not students:
        print("暂无学生成绩记录")
        return
    
    total = sum(students.values())
    average = total / len(students)
    print(f"平均分: {average:.2f}")

def find_student(students):
    name = input("请输入要查找的学生姓名: ")
    if name in students:
        print(f"{name}的成绩是: {students[name]}")
    else:
        print(f"未找到学生 {name}")

def delete_student(students):
    name = input("请输入要删除的学生姓名: ")
    if name in students:
        del students[name]
        print(f"成功删除学生 {name}")
    else:
        print(f"未找到学生 {name}")

def main():
    students = {}
    
    while True:
        display_menu()
        choice = input("请选择操作(1-6): ")
        
        if choice == '1':
            add_student(students)
        elif choice == '2':
            view_all_students(students)
        elif choice == '3':
            calculate_average(students)
        elif choice == '4':
            find_student(students)
        elif choice == '5':
            delete_student(students)
        elif choice == '6':
            print("感谢使用，再见！")
            break
        else:
            print("无效选择，请重新输入")

# 运行程序
if __name__ == "__main__":
    main()

2. 数据分析小工具

# 数据分析小工具
def analyze_data(data):
    """分析数据并返回统计信息"""
    if not data:
        return "数据为空"
    
    stats = {}
    stats['count'] = len(data)
    stats['sum'] = sum(data)
    stats['average'] = stats['sum'] / stats['count']
    stats['max'] = max(data)
    stats['min'] = min(data)
    stats['range'] = stats['max'] - stats['min']
    
    # 计算中位数
    sorted_data = sorted(data)
    mid = stats['count'] // 2
    if stats['count'] % 2 == 0:
        stats['median'] = (sorted_data[mid-1] + sorted_data[mid]) / 2
    else:
        stats['median'] = sorted_data[mid]
    
    return stats

def display_stats(stats):
    """显示统计信息"""
    if isinstance(stats, str):
        print(stats)
        return
    
    print("\n=== 数据分析结果 ===")
    print(f"数据个数: {stats['count']}")
    print(f"总和: {stats['sum']}")
    print(f"平均值: {stats['average']:.2f}")
    print(f"最大值: {stats['max']}")
    print(f"最小值: {stats['min']}")
    print(f"极差: {stats['range']}")
    print(f"中位数: {stats['median']}")

# 测试数据
test_data = [23, 45, 67, 12, 89, 34, 56, 78, 90, 11]

# 分析数据
results = analyze_data(test_data)
display_stats(results)

# 空数据测试
empty_data = []
empty_results = analyze_data(empty_data)
display_stats(empty_results)

题目3：综合编程

编写一个程序，实现以下功能：

  1. 生成1-100之间的随机整数列表（包含10个数字）
  2. 找出列表中的素数和合数
  3. 计算素数和合数的平均值
  4. 找出最大的素数和最小的合数
# 答案
import random

def is_prime(n):
    """判断一个数是否为素数"""
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

# 生成随机数列表
numbers = [random.randint(1, 100) for _ in range(10)]
print(f"生成的随机数: {numbers}")

# 分类素数和合数
primes = []
composites = []

for num in numbers:
    if is_prime(num):
        primes.append(num)
    else:
        composites.append(num)

print(f"素数: {primes}")
print(f"合数: {composites}")

# 计算平均值
prime_avg = sum(primes) / len(primes) if primes else 0
composite_avg = sum(composites) / len(composites) if composites else 0

print(f"素数平均值: {prime_avg:.2f}")
print(f"合数平均值: {composite_avg:.2f}")

# 找出最大的素数和最小的合数
max_prime = max(primes) if primes else "无"
min_composite = min(composites) if composites else "无"

print(f"最大素数: {max_prime}")
print(f"最小合数: {min_composite}")
