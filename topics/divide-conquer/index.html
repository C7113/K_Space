            // nums[i]åˆ°nums[mid]éƒ½å¤§äºnums[j]ï¼Œå¢åŠ é€†åºå¯¹æ•°
            count += (mid - i + 1);
        }
    }

    // å¤„ç†å‰©ä½™å…ƒç´ 
    while (i <= mid) {
        temp[k++] = nums[i++];
    }
    while (j <= right) {
        temp[k++] = nums[j++];
    }

    // å°†åˆå¹¶åçš„ç»“æœå¤åˆ¶å›åŸæ•°ç»„
    for (int i = 0; i < k; i++) {
        nums[left + i] = temp[i];
    }

    return count;
}

int mergeSortAndCount(vector<int>& nums, int left, int right) {
    if (left >= right) return 0;

    // åˆ†è§£ï¼šå°†æ•°ç»„åˆ†æˆä¸¤åŠ
    int mid = left + (right - left) / 2;

    // è§£å†³ï¼šé€’å½’è®¡ç®—å·¦å³ä¸¤åŠçš„é€†åºå¯¹æ•°
    int leftCount = mergeSortAndCount(nums, left, mid);
    int rightCount = mergeSortAndCount(nums, mid + 1, right);

    // åˆå¹¶ï¼šè®¡ç®—è·¨è¶Šä¸­ç‚¹çš„é€†åºå¯¹æ•°
    int crossCount = mergeAndCount(nums, left, mid, right);

    // è¿”å›æ€»é€†åºå¯¹æ•°
    return leftCount + rightCount + crossCount;
}

int countInversions(vector<int>& nums) {
    if (nums.size() <= 1) return 0;
    vector<int> temp = nums;  // åˆ›å»ºå‰¯æœ¬é¿å…ä¿®æ”¹åŸæ•°ç»„
    return mergeSortAndCount(temp, 0, nums.size() - 1);
}
