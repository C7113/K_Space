STL基本概念

什么是STL

STL（Standard Template Library，标准模板库）是C++标准库的重要组成部分，它提供了一套高度可复用的组件，包括容器、算法和迭代器。

STL的特点

  • 高效性：STL组件经过高度优化，性能优异
  • 通用性：使用模板技术，支持多种数据类型
  • 可扩展性：可以方便地扩展和定制
  • 标准化：遵循C++标准，具有良好的可移植性

STL的组成

STL主要由以下几部分组成:

1. 容器(Container)
   - 顺序容器: vector, list, deque
   - 关联容器: set, map, multiset, multimap
   - 容器适配器: stack, queue, priority_queue

2. 算法(Algorithm)
   - 查找算法: find, binary_search
   - 排序算法: sort, stable_sort
   - 数值算法: accumulate, inner_product

3. 迭代器(Iterator)
   - 输入迭代器、输出迭代器
   - 前向迭代器、双向迭代器、随机访问迭代器

4. 函数对象(Function Object)
   - 算术函数对象
   - 关系函数对象
   - 逻辑函数对象

5. 适配器(Adapter)
   - 容器适配器
   - 函数适配器
   - 迭代器适配器

STL的重要性

STL是现代C++编程的基础，掌握STL对于提高编程效率和代码质量具有重要意义。

学习STL的意义

  • 提高开发效率：使用现成的高效组件，避免重复造轮子
  • 提升代码质量：STL组件经过严格测试，稳定性高
  • 增强程序性能：STL组件经过优化，性能优异
  • 为后续学习打基础：为学习更高级的C++特性和算法打下基础

STL容器

容器的概念

容器是用来存储和管理数据的对象，STL提供了多种不同特性的容器以满足不同的需求。

容器的分类

  • 顺序容器：元素按线性顺序排列
  • 关联容器：通过键值对存储数据
  • 容器适配器：基于其他容器实现特定接口

常用容器介绍

以下是一些常用的STL容器及其特点：

vector（动态数组）

#include <iostream>
#include <vector>
using namespace std;

int main() {
    // 创建vector容器
    vector<int> vec;

    // 添加元素
    vec.push_back(10);
    vec.push_back(20);
    vec.push_back(30);

    // 访问元素
    cout << "第一个元素: " << vec[0] << endl;
    cout << "容器大小: " << vec.size() << endl;

    // 遍历容器
    cout << "所有元素: ";
    for (int i = 0; i < vec.size(); i++) {
        cout << vec[i] << " ";
    }
    cout << endl;

    return 0;
}

stack（栈）

#include <iostream>
#include <stack>
using namespace std;

int main() {
    // 创建stack容器
    stack<int> stk;

    // 入栈操作
    stk.push(10);
    stk.push(20);
    stk.push(30);

    // 访问栈顶元素
    cout << "栈顶元素: " << stk.top() << endl;
    cout << "栈的大小: " << stk.size() << endl;

    // 出栈操作
    while (!stk.empty()) {
        cout << "弹出: " << stk.top() << endl;
        stk.pop();
    }

    return 0;
}

STL算法

算法的概念

STL算法是一组用于处理容器中数据的函数模板，它们通过迭代器访问容器中的元素。

算法的特点

  • 通用性：可以用于任何支持相应迭代器的容器
  • 高效性：经过优化实现，性能优异
  • 一致性：具有统一的接口和使用方式

常用算法介绍

以下是一些常用的STL算法：

sort（排序算法）

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    // 创建vector容器并初始化
    vector<int> vec = {5, 2, 8, 1, 9, 3};

    // 输出排序前的元素
    cout << "排序前: ";
    for (int i = 0; i < vec.size(); i++) {
        cout << vec[i] << " ";
    }
    cout << endl;

    // 使用sort算法排序
    sort(vec.begin(), vec.end());

    // 输出排序后的元素
    cout << "排序后: ";
    for (int i = 0; i < vec.size(); i++) {
        cout << vec[i] << " ";
    }
    cout << endl;

    return 0;
}

find（查找算法）

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    // 创建vector容器并初始化
    vector<int> vec = {10, 20, 30, 40, 50};

    // 使用find算法查找元素
    vector<int>::iterator it = find(vec.begin(), vec.end(), 30);

    // 判断是否找到
    if (it != vec.end()) {
        cout << "找到了元素 30，位置: " << (it - vec.begin()) << endl;
    } else {
        cout << "未找到元素 30" << endl;
    }

    return 0;
}

STL迭代器

迭代器的概念

迭代器是一种广义的指针，它提供了一种统一的方式来访问容器中的元素，而不需要了解容器的内部实现。

迭代器的分类

迭代器类型 功能 支持的操作
输入迭代器 只能读取数据 ++it, *it, it1 == it2, it1 != it2
输出迭代器 只能写入数据 ++it, *it
前向迭代器 可读写，可向前移动 ++it, *it, it1 == it2, it1 != it2
双向迭代器 可读写，可前后移动 ++it, --it, *it, it1 == it2, it1 != it2
随机访问迭代器 可读写，可随机访问 ++it, --it, it+n, it-n, it1-it2, it1

迭代器的使用

迭代器是连接容器和算法的桥梁，通过迭代器可以访问和操作容器中的元素。

使用迭代器遍历容器

#include <iostream>
#include <vector>
using namespace std;

int main() {
    // 创建vector容器并初始化
    vector<int> vec = {1, 2, 3, 4, 5};

    // 使用迭代器遍历容器
    cout << "使用迭代器遍历: ";
    for (vector<int>::iterator it = vec.begin(); it != vec.end(); ++it) {
        cout << *it << " ";
    }
    cout << endl;

    // 使用const_iterator遍历（只读）
    cout << "使用const_iterator遍历: ";
    for (vector<int>::const_iterator it = vec.begin(); it != vec.end(); ++it) {
        cout << *it << " ";
    }
    cout << endl;

    // C++11范围for循环（最简洁的方式）
    cout << "使用范围for循环: ";
    for (int value : vec) {
        cout << value << " ";
    }
    cout << endl;

    return 0;
}

示例1：学生成绩管理

使用vector容器和STL算法管理学生成绩：

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    // 使用vector存储学生成绩
    vector<int> scores = {85, 92, 78, 96, 88, 76, 89, 94, 82, 90};

    // 输出所有成绩
    cout << "学生成绩: ";
    for (int score : scores) {
        cout << score << " ";
    }
    cout << endl;

    // 计算最高分和最低分
    int maxScore = *max_element(scores.begin(), scores.end());
    int minScore = *min_element(scores.begin(), scores.end());
    cout << "最高分: " << maxScore << endl;
    cout << "最低分: " << minScore << endl;

    // 计算平均分
    int sum = 0;
    for (int score : scores) {
        sum += score;
    }
    double average = (double)sum / scores.size();
    cout << "平均分: " << average << endl;

    // 排序成绩
    sort(scores.begin(), scores.end());

    for (int score : scores) {
        cout << score << " ";
    }

    // 查找特定成绩
    int target = 90;
    vector<int>::iterator it = find(scores.begin(), scores.end(), target);

        cout << "未找到成绩 " << target << endl;
    }

示例2：括号匹配检查

使用stack容器检查括号是否匹配：

#include <iostream>
#include <stack>
#include <string>
using namespace std;

bool isBalanced(const string& expr) {
    stack<char> s;

    for (char c : expr) {
        if (c == '(' || c == '[' || c == '{') {
            // 遇到开括号，入栈
            s.push(c);
        } else if (c == ')' || c == ']' || c == '}') {
            // 遇到闭括号
            if (s.empty()) {
                // 栈为空，不匹配
                return false;
            }

            char top = s.top();
            s.pop();

            // 检查括号是否匹配
            if ((c == ')' && top != '(') ||
                (c == ']' && top != '[') ||
                (c == '}' && top != '{')) {
                return false;
            }
        }
    }

    // 检查栈是否为空
    return s.empty();
}

int main() {
    string expr1 = "{[()()]}";
    string expr2 = "{[()]}";
    string expr3 = "{[()])";

    cout << expr1 << " 是否匹配: " << (isBalanced(expr1) ? "是" : "否") << endl;
    cout << expr2 << " 是否匹配: " << (isBalanced(expr2) ? "是" : "否") << endl;
    cout << expr3 << " 是否匹配: " << (isBalanced(expr3) ? "是" : "否") << endl;

    return 0;
}

#include <iostream>
#include <vector>
using namespace std;

// 在此处添加代码
class SimpleQueue {
private:
    vector<int> data;

public:
    // 入队操作
    void enqueue(int value) {
        data.push_back(value);
    }

    // 出队操作
    int dequeue() {
        if (isEmpty()) {
            cout << "队列为空!" << endl;
            return -1;
        }
        int value = data.front();
        data.erase(data.begin());
        return value;
    }

    // 检查队列是否为空
    bool isEmpty() {
        return data.empty();
    }

    // 获取队列大小
    int size() {
        return data.size();
    }

    // 显示队列内容
    void display() {
        cout << "队列内容: ";
        for (int value : data) {
            cout << value << " ";
        }
        cout << endl;
    }
};

int main() {
    SimpleQueue queue;

    // 测试队列操作
    queue.enqueue(10);
    queue.enqueue(20);
    queue.enqueue(30);

    queue.display();

    cout << "出队元素: " << queue.dequeue() << endl;
    cout << "出队元素: " << queue.dequeue() << endl;

    queue.display();

    return 0;
}

练习2：使用algorithm算法处理数据

使用STL算法处理整数数组：

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

// 在此处添加代码
int main() {
    vector<int> numbers = {3, 1, 4, 1, 5, 9, 2, 6, 5, 3};

    // 输出原始数据
    cout << "原始数据: ";
    for (int n : numbers) {
        cout << n << " ";
    }
    cout << endl;

    // 1. 查找元素
    vector<int>::iterator it = find(numbers.begin(), numbers.end(), 5);
    if (it != numbers.end()) {
        cout << "找到了元素5，位置: " << (it - numbers.begin()) << endl;
    }

    // 2. 统计元素出现次数
    int count5 = count(numbers.begin(), numbers.end(), 5);
    cout << "元素5出现次数: " << count5 << endl;

    // 3. 排序
    sort(numbers.begin(), numbers.end());
    cout << "排序后: ";
    for (int n : numbers) {
        cout << n << " ";
    }
    cout << endl;

    // 4. 去除重复元素
    vector<int>::iterator newEnd = unique(numbers.begin(), numbers.end());
    numbers.erase(newEnd, numbers.end());
    cout << "去重后: ";
    for (int n : numbers) {
        cout << n << " ";
    }
    cout << endl;

    return 0;
}

练习3：使用stack实现表达式求值

使用stack容器实现简单表达式的求值：

#include <iostream>
#include <stack>
#include <string>
using namespace std;

// 在此处添加代码
int evaluateExpression(const string& expr) {
    stack<int> operands;
    stack<char> operators;

    for (int i = 0; i < expr.length(); i++) {
        char c = expr[i];

        if (isdigit(c)) {
            // 处理数字
            int num = 0;
            while (i < expr.length() && isdigit(expr[i])) {
                num = num * 10 + (expr[i] - '0');
                i++;
            }
            i--;  // 回退一位，因为for循环会自增
            operands.push(num);
        } else if (c == '+' || c == '-' || c == '*' || c == '/') {
            // 处理操作符
            while (!operators.empty() &&
                ((c == '+' || c == '-') && !operators.empty())) {
                // 简化处理，这里只处理同级运算
                if (operators.top() == '+' || operators.top() == '-') {
                    break;
                }
                char op = operators.top();
                operators.pop();

                int b = operands.top();
                operands.pop();
                int a = operands.top();
                operands.pop();

                switch (op) {
                    case '+':
                        operands.push(a + b);
                        break;
                    case '-':
                        operands.push(a - b);
                        break;
                    case '*':
                        operands.push(a * b);
                        break;
                    case '/':
                        operands.push(a / b);
                        break;
                }
            }
            operators.push(c);
        }
    }

    // 处理剩余的操作符
    while (!operators.empty()) {
        char op = operators.top();
        operators.pop();

        int b = operands.top();
        operands.pop();
        int a = operands.top();
        operands.pop();

        switch (op) {
            case '+':
                operands.push(a + b);
                break;
            case '-':
                operands.push(a - b);
                break;
            case '*':
                operands.push(a * b);
                break;
            case '/':
                operands.push(a / b);
                break;
        }
    }

    return operands.top();
}

int main() {
    string expr = "3+5*2";
    cout << expr << " = " << evaluateExpression(expr) << endl;

    return 0;
}
