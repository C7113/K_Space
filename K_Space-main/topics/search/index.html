    bool isValid(int row, int col) {

        for (int i = 0; i < row; i++) {

        for (int i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++) {

        }

        return true;
    }

    // DFSå›æº¯æ±‚è§£Nçš‡åé—®é¢˜
    void solve(int row) {
        if (row == n) {
            // æ‰¾åˆ°ä¸€ä¸ªè§£
            solutions.push_back(board);
            return;
        }

        // åœ¨å½“å‰è¡Œçš„æ¯ä¸€åˆ—å°è¯•æ”¾ç½®çš‡å
        for (int col = 0; col < n; col++) {

                board[row][col] = 'Q';

                // é€’å½’å¤„ç†ä¸‹ä¸€è¡Œ

                // å›æº¯ï¼Œç§»é™¤çš‡å
                board[row][col] = '.';
            }
        }
    }

    vector<vector<string>> solveNQueens(int n) {
        this->n = n;
        solutions.clear();
        board = vector<string>(n, string(n, '.'));

        solve(0);

    }
